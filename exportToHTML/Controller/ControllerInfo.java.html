1    package Controller; 
2     
3    import javafx.fxml.FXML; 
4    import javafx.scene.control.TextArea; 
5     
6    import java.text.DateFormat; 
7    import java.text.SimpleDateFormat; 
8    import java.util.Date; 
9     
10   /** 
11    * Отвечает за вывод информации на екран 
12    */ 
13   public class ControllerInfo { 
14    
15       @FXML private TextArea textAreaInfo; 
16    
17       Controller mainController = null; 
18    
19       // Способ передать ссылку на клас-контроллер в главный контроллер 
20       // Создаем статичесскую переменную в которой будем хранить ссылку на созданый екземпляр контролера 
21       public static ControllerInfo instance = null; 
22       // Как только вызывается конструктор класа - "запоминаем" ссылку на него в статическое поле. 
23       // Таким образром мы сможем получить ссылку на этот контроллер в дальнейшем 
24       public ControllerInfo(){ 
25           instance = this; 
26       } 
27    
28       /** 
29        * Инициализируем контроллер передачей в него ссылки на главный контроллер 
30        * @param controller ссылка на контроллер 
31        */ 
32       public void init(Controller controller){ 
33           mainController = controller; 
34       } 
35    
36       /** 
37        * выводит текст в окно и добавляет временной штамп 
38        * @param info 
39        */ 
40       public void showInfo(String info){ 
41           DateFormat dateFormat = new SimpleDateFormat("HH:mm:ss"); 
42           Date date = new Date(); 
43    
44           textAreaInfo.setText(textAreaInfo.getText() + "\n" + dateFormat.format(date).toString()+ " " + info); 
45    
46       } 
47    
48    
49   }
